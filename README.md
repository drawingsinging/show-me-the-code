puts ARGF.read.each_line.map {|line| line[/([^\s]+?)\.md/]; $1}.compact
  .keep_if {|item| !(%w[rule README readme].include?(item))}
  .reduce(Hash.new(0)) {|h, e| h[e]+=1; h}.to_a

  .map.with_index {|(a,b),idx| "|##{idx}|[#{a}]@#{names[a]}|#{b}|"}.join("\n")'
